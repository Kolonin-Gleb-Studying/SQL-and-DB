DELIMITER //

CREATE TRIGGER log_change
AFTER UPDATE
ON hw2_news
FOR EACH ROW
BEGIN
    INSERT INTO hw2_logging 
    VALUES (0, NEW.id, OLD.title, NEW.title, OLD.info, NEW.info, NEW.dt, OLD.dt,  NOW());
END //

DELIMITER ;
